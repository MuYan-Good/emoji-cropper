## 4. TypeScript 规范

### 4.1 类型导入

类型导入必须使用 `import type` 语法：

```typescript
// ✅ 正确
import type { EmojiItem } from '../types/emoji'
import { cropImage } from '../utils/imageCropper'
import type { GridConfig } from '../types/image'

// ❌ 错误
import { EmojiItem, GridConfig } from '../types/emoji'
```

### 4.2 interface vs type

| 场景 | 使用 | 示例 |
|------|------|------|
| 对象结构描述 | `interface` | `interface EmojiItem { id: string; name: string }` |
| 联合类型、交叉类型、工具类型 | `type` | `type ExportFormat = 'png' \| 'gif'` |
| 函数签名 | `type` | `type CropFn = (image: HTMLImageElement, config: GridConfig) => Promise<Blob>` |

### 4.3 禁止的写法

```typescript
// ❌ 禁止使用 any
function process(data: any) { }

// ✅ 使用 unknown 或具体类型
function process(data: unknown) { }
function process(data: Blob) { }

// ❌ 禁止非空断言
const name = item.name!
const el = document.getElementById('canvas')!

// ✅ 使用可选链或类型守卫
const name = item.name ?? 'default'
const el = document.getElementById('canvas')
if (!el) throw new Error('Canvas not found')

// ❌ 禁止 as 类型断言（除非不可避免）
const config = data as GridConfig

// ✅ 使用类型守卫
function isGridConfig(data: unknown): data is GridConfig { ... }
```

## 5. React 规范

### 5.1 组件文件结构

每个组件文件必须按以下顺序组织：

```typescript
// 1. 类型导入
import type { ChangeEvent } from 'react'

// 2. React 导入
import { useState, useCallback } from 'react'

// 3. 第三方库导入
import { useDropzone } from 'react-dropzone'

// 4. 内部模块导入（types → utils → hooks → components）
import type { EmojiItem } from '../types/emoji'
import { DEFAULT_OUTPUT_SIZE } from '../utils/constants'
import { useImageUpload } from '../hooks/useImageUpload'

// 5. 类型定义（组件 props）
interface ImageUploaderProps {
  onImageLoad: (file: File) => void
  maxSize?: number
}

// 6. 组件定义
export function ImageUploader({ onImageLoad, maxSize = 10 * 1024 * 1024 }: ImageUploaderProps) {
  const [isDragging, setIsDragging] = useState(false)

  const handleDrop = useCallback(
    (acceptedFiles: File[]) => {
      if (acceptedFiles.length > 0) {
        onImageLoad(acceptedFiles[0])
      }
    },
    [onImageLoad],
  )

  return (
    <div>
      {/* JSX */}
    </div>
  )
}
```

### 5.2 组件设计原则

- **函数式组件**：禁止使用 class 组件
- **导出方式**：使用命名导出 `export function Xxx`，禁止默认导出
- **Props 类型**：必须定义 interface，禁止内联类型
- **单一职责**：一个组件只做一件事，超过 150 行考虑拆分

### 5.3 Hooks 使用规范

```typescript
// ✅ 自定义 Hook 必须以 use 开头
export function useSpriteCrop(image: HTMLImageElement | null, config: GridConfig) {
  const [results, setResults] = useState<CropResult[]>([])
  // ...
  return { results, isCropping, crop }
}

// ✅ Hook 调用顺序必须稳定，不能放在条件语句中
function Component() {
  const [isOpen, setIsOpen] = useState(false)
  const data = useCustomHook() // ✅ 始终调用

  return <div />
}

// ❌ 禁止条件调用 Hook
function Component() {
  if (isOpen) {
    const data = useCustomHook() // ❌ 违反 Hook 规则
  }
  return <div />
}
```

### 5.4 导入排序规则

导入分为 4 组，组间空行分隔：

```typescript

import { useState, useCallback } from 'react'
```

### 7.2 分支命名规范

格式：`<type>/<description>`

| type | 用途 | 示例 |

## 8. 代码注释规范

### 8.1 何时写注释 vs 何时不写

| 场景 | 是否写注释 | 说明 |
|------|-----------|------|
| 代码本身能清楚表达意图 | ❌ 不写 | 好的命名和类型比注释更有价值 |
| 解释"为什么"而非"做了什么" | ✅ 写 | 帮助后续维护者理解设计决策 |
| 复杂算法或非直觉逻辑 | ✅ 写 | 简要说明算法思路或引用 |
| 公共 API / 工具函数 | ✅ 写 | 使用 JSDoc 说明参数、返回值、用法 |
| TODO / FIXME | ✅ 写 | 标记待办或已知问题 |
| 临时方案 / Hack | ✅ 写 | 说明原因和计划移除时间 |

### 8.2 注释风格

```typescript
// ✅ 解释"为什么"
// GIF 帧延迟小于 20ms 会导致部分浏览器播放过快，因此设置下限
const MIN_FRAME_DELAY = 20

// ❌ 解释"做了什么"（代码本身已经说明）
// 设置最小帧延迟为 20ms
const MIN_FRAME_DELAY = 20

// ✅ JSDoc 用于公共 API
/**
 * 将雪碧图按网格裁剪为单个表情图片

 */
export async function cropSprite(
  image: HTMLImageElement,
  config: GridConfig,
  outputSize: number,
): Promise<Blob[]> { ... }
```

## 9. 错误处理规范

### 9.1 工具函数层（utils/）

```typescript
// ✅ 纯函数抛出明确的 Error
export function validateFile(file: File): void {
  if (!SUPPORTED_TYPES.includes(file.type)) {
    throw new Error(`不支持的文件格式: ${file.type}，仅支持 PNG 和 GIF`)
  }
  if (file.size > MAX_FILE_SIZE) {
    throw new Error(`文件大小超过限制: ${(file.size / 1024 / 1024).toFixed(1)}MB，最大支持 10MB`)
  }
}

// ✅ 异步函数返回 Result 模式或抛出 Error
export async function cropSprite(
  image: HTMLImageElement,
  config: GridConfig,
  outputSize: number,
): Promise<Blob[]> {
  if (config.rows <= 0 || config.cols <= 0) {
    throw new Error(`行列数必须大于 0，当前: ${config.rows}×${config.cols}`)
  }
  // ...
}
```

### 9.2 Hooks 层

```typescript
// ✅ Hook 捕获错误并暴露给组件
export function useSpriteCrop() {
  const [error, setError] = useState<string | null>(null)
  const [isCropping, setIsCropping] = useState(false)

  const crop = useCallback(async (image: HTMLImageElement, config: GridConfig) => {
    setIsCropping(true)
    setError(null)
    try {
      const results = await cropSprite(image, config, DEFAULT_OUTPUT_SIZE)
      return results
    } catch (err) {
      const message = err instanceof Error ? err.message : '裁剪失败，请重试'
      setError(message)
      return []
    } finally {
      setIsCropping(false)
    }
  }, [])

  return { crop, isCropping, error }
}
```

### 9.3 组件层

```tsx
// ✅ 组件展示用户友好的错误提示
{error && (
  <div className="rounded-lg bg-red-50 p-4 text-sm text-red-600">
    {error}
  </div>
)}
```

### 9.4 用户提示规则

- 错误提示使用**中文**，面向普通用户
- 禁止暴露技术细节（如堆栈信息、内部变量名）
- 提供可操作的建议（如"请上传 PNG 或 GIF 格式的图片"）

### 14.1 合法依赖方向

```
components/ ──→ hooks/ ──→ utils/ ──→ types/
     │              │
     └──────────────┴──→ types/
```

### 14.2 依赖规则

| 源模块 | 可依赖 | 禁止依赖 |
|--------|--------|----------|
| `components/` | `hooks/`、`types/`、第三方库 | `utils/`（通过 hooks 间接使用） |
| `hooks/` | `utils/`、`types/`、第三方库 | `components/` |
| `utils/` | `types/`、第三方库 | `hooks/`、`components/` |
| `types/` | 无 | 任何业务模块 |

### 14.3 禁止的依赖方向

```
❌ utils/ → hooks/
❌ utils/ → components/
❌ hooks/ → components/
❌ types/ → 任何业务模块
❌ 跨层级跳过（如 components/ 直接 import utils/）
```

## 15. 安全编码规范

### 15.1 文件上传安全

```typescript
// ✅ 严格校验文件类型（检查 MIME type，不仅依赖扩展名）
const ACCEPTED_TYPES = ['image/png', 'image/gif']

function validateFileType(file: File): boolean {
  return ACCEPTED_TYPES.includes(file.type)
}

// ✅ 限制文件大小
const MAX_FILE_SIZE = 10 * 1024 * 1024 // 10MB
if (file.size > MAX_FILE_SIZE) {
  throw new Error('文件大小超过 10MB 限制')
}
```

### 15.2 XSS 防护

```tsx
// ✅ React 默认转义 JSX 中的文本内容
<span>{userInput}</span>  // 安全，自动转义

// ❌ 禁止使用 dangerouslySetInnerHTML（本项目不需要）
<div dangerouslySetInnerHTML={{ __html: userInput }} />

// ❌ 禁止拼接用户输入到 URL 或属性中
<img src={`data:${userInput}`} />
```

### 15.3 内存安全

```typescript
// ✅ 处理大图片后释放资源
const canvas = document.createElement('canvas')
// ... 使用 canvas 处理图片 ...
canvas.width = 0
canvas.height = 0
// canvas 引用超出作用域后会被 GC 回收

// ✅ 及时释放 Object URL
const url = URL.createObjectURL(blob)
// 使用完毕后
URL.revokeObjectURL(url)
```

### 15.4 数据隐私

- 所有图片处理在浏览器端完成，禁止将用户图片上传到任何服务器
- 禁止在代码中硬编码 API Key、密钥等敏感信息
- 禁止使用 `localStorage` 存储用户图片数据（仅可用于存储用户偏好设置）
